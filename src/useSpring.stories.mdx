# useSpring

To get a spring function to interpolate.

## Basic

<Story name="Basic">
  <Test tr={{ left: [0, 150, 'px'] }} />
  <TestComplete tr={{ left: [0, 150, 'px'] }} />
</Story>

```tsx
import React from 'react'
import useSpring, { interpolate } from './useSpring'

export const Test = () => {
  const { value, start } = useSpring({ auto: false })
  return (
    <>
      <button style={{ zIndex: 99, marginBottom: 50 }} onClick={() => start()}>
        start
      </button>
      <div style={{ zIndex: 0, position: 'relative' }}>
        <span style={{ position: 'relative', display: 'block', width: 150, height: 50, ...interpolate({ left: [0, 150, 'px'] }, value) }}>
          Hello
        </span>
      </div>
    </>
  )
}
```
